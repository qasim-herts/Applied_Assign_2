#Funcation which the file name as argument and return two dataframes 
def ingest_manipulate(file_name=r'D:\New folder\a_Data.csv'):
    """Return two Dataframes  
 
    Read the file name as Argument
 
    Parameters
    ------------
        file_name
    Return
    -----------
        DataFrame1(df1) : Return Years as a columns
            DataFrame1(df2) : Return countries as a columns. """
    df = pd.read_csv(file_name)
    df.drop(['Country Name', 'Series Name'], axis=1, inplace=True)
    df1 = df.groupby(['Country Code', 'Series Code']).sum()
    return df1, df1.T
